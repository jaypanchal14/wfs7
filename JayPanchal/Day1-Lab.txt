Program to reverse each word of set and then sort it.

	{	String arr[] = new String[]{"apple","banana","sheet","dog","cat"};

package com.hsbc.pack;
/*
	Implementation of selection sort
	logic:
	finding smallest numbers and swapping with respective index. 
*/
class Exer11
{	public static void main(String args[])
	{	
		int[] arr = {2,6,9,3,4};
		System.out.println("Before selection sort");
		for(int i:arr)
		{	System.out.println(i+"");
		}
		for(int i=0;;i<arr.length-1;i++)
		{	int index = i;
			for(int j=i+1;j<arr.length;j++)
			{	if(arr[j]<arr[index])
				{  index = j;     }
			}
			// Swapping between i and smallestNumber
			int smallNum = arr[index];
			arr[index] = arr[i];
			arr[i] = smallNum;
		}
		System.out.println("After selection sort");
		for(int i:arr)
		{	System.out.println(i+"");
		}
	}
}

package com.hsbc.pack;
/*
	To find the total sum of cost after passing which product
	we are goint to buy and how much.	
*/
class Exer12
{	public static void main(String args[])
	{	
		float PRODUCT1 = 22.50f;
		float PRODUCT2 = 44.50f;
		float PRODUCT3 = 9.98f;
		int product = Integer.parseInt(args[0]);
		int quantity = Integer.parseInt(args[1]);
		float price;
		switch(product)
		{	case 1:
			price = PRODUCT1;
			break;
			case 2:
			price = PRODUCT2;
			break;
			case 3:
			price = PRODUCT3;
			break;
			default:
			price = 5
			System.out.println("Invalid product-number");
			break;
		}
		if(price>0 && price<4)
		{float sum = price*quantity;
		System.out.println("Total value:"+sum);	}
	}
}
